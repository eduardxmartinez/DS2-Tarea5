    public static void inicializar(int w[][]) {

        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                w[i][j] = 0;
            }
        }
    }

    public static void despliega(int w[][]) {

        for (int j = 0; j < 3; j++) {
            for (int k = 0; k < 3; k++) {
                System.out.print(w[j][k]);
                System.out.print(" ");
            }
            System.out.println("");
        }
        System.out.println("---------------");
    }

    public static boolean verifica(int w[][], int i) {
        if (((w[0][0] == i) && (w[0][1] == i) && (w[0][2] == i))
                || ((w[1][0] == i) && (w[1][1] == i) && (w[1][2] == i))
                || ((w[2][0] == i) && (w[2][1] == i) && (w[2][2] == i))
                || ((w[0][0] == i) && (w[1][0] == i) && (w[2][0] == i))
                || ((w[0][1] == i) && (w[1][1] == i) && (w[2][1] == i))
                || ((w[0][2] == i) && (w[1][2] == i) && (w[2][2] == i))
                || ((w[0][0] == i) && (w[1][1] == i) && (w[2][2] == i))
                || ((w[0][2] == i) && (w[1][1] == i) && (w[2][0] == i))) {
            System.out.println("Gana Jugador " + i);
            return true;
        }
        return false;
    }

    public static void jugador1(int w[][]) {
        Scanner sc = new Scanner(System.in);
